void getDetails(EMP e[], int n)
{
	int i;

	printf("\nEnter Employee Details\n");
	for(i=0;i<n;i++)

		printf("\nName: ");
		scanf("%s",e[i].eName);

		scanf("%f",&e[i].eSal);

		scanf("%s",&e[i].eLname);

}

int searchId(EMP e[],int n,int id)
{
    for(int i=0;i<n;i++)
    {
        if(e[i].eId==id)
              return i;
    }
    return -1;
}
void printId(EMP e[],int n,int id)
{
    int search= searchId(e,n,id);
    if(search!=1)
    {

}
